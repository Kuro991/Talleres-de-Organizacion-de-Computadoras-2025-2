section .data
char db 0
newline db 0xA

section .text
global _start
_start:

mov al, 0x0D ; n√∫mero base

shl al, 3 ; 0x0D << 3 = 0x68
ror al, 1 ; 0x68 ror1 = 0x34
shl al, 1 ; 0x34 << 1 = 0x68
sub al, 1 ; 0x68 - 1 = 0x67 ('g')

mov [char], al

mov eax, 4
mov ebx, 1
mov ecx, char

mov edx, 1
int 0x80

mov eax, 4
mov ebx, 1
mov ecx, newline
mov edx, 1
int 0x80

mov eax, 1
xor ebx, ebx
int 0x80
