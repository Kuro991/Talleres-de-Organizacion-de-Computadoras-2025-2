; Entrada: [X], [Y]
MOV EAX, [X]
MOV EBX, [Y]
CMP EAX, EBX
JLE YA_ORDENADOS ; si X <= Y, ya están en orden ascendente

; Si llegamos aquí, X > Y -> intercambiar
; swap:
MOV ECX, EAX
MOV EAX, EBX
MOV EBX, ECX
; guardar de vuelta
MOV [X], EAX
MOV [Y], EBX
JMP FIN

YA_ORDENADOS:
; nada por hacer
FIN:
; fin
