; Salidas: mensajes (IGUALES, A_MAYOR, B_MAYOR, NEGATIVO)

TEST EAX, EAX
JS A_NEG ; si A < 0 (Sign flag), saltar
MOV EBX, [B] ; EBX = B
TEST EBX, EBX
JS B_NEG ; si B < 0
; comparar A y B (comparación de enteros con CMP)
CMP EAX, EBX
JE SON_IGUALES ; ZF = 1 -> iguales
JG A_MAYOR ; SF/OF combinados por JG -> A > B (signed)
JL B_MAYOR ; A < B

JMP FIN

JMP FIN

; imprimir "A y B son iguales"
